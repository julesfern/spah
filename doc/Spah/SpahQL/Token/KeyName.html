def self.parse_at(i, query)
     valid = /[\w\d_-]/; j=i; token = "";
     m = nil
     while(m = query[j,1].match(valid)) do
       token += m[0]
       j += 1
     end#while
     return (token.length > 0)? [j, self.new(token)] : nil;
  end
